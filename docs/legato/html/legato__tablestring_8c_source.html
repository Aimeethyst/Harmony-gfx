123 uint32_t _leTableString_Length(const leTableString* _this)
124 {
125  const leStringTable* tbl = leGetState()->stringTable;
126  leStringInfo info;
127 
128  LE_ASSERT_THIS();
129 
130  if(_this->index == LE_STRING_NULLIDX)
131  return 0;
132 
133  info.stringIndex = _this->index;
134  info.languageID = leGetState()->languageID;
135 
136  if(leStringTable_StringLookup(tbl, &info) == LE_FAILURE)
137  return 0;
138 
139  if(leStringTable_GetStringLength(tbl, &info) == LE_FAILURE)
140  return 0;
141 
142  return info.length;
143 }

152 leChar _leTableString_CharAt(const leTableString* _this,
153  uint32_t idx)
154 {
155  leFont* fnt = NULL;
156  uint32_t i, j;
157  uint32_t codePoint;
158  uint32_t offset = 0;
159  const leStringTable* tbl = leGetState()->stringTable;
160  uint32_t lang = leGetState()->languageID;
161  leStringInfo info;
162 
163  LE_ASSERT_THIS();
164 
165  if(tbl == NULL)
166  return 0;
167 
168  fnt = leStringTable_GetStringFont(tbl, _this->index, lang);
169 
170  if(fnt == NULL)
171  return 0;
172 
173  info.stringIndex = _this->index;
174  info.languageID = lang;
175 
176  if(leStringTable_StringLookup(tbl, &info) == LE_FAILURE)
177  return LE_FAILURE;
178 
179  j = 0;
180 
181  for(i = 0; i < info.dataSize;)
182  {
183  if(leDecodeCodePoint(tbl->encodingMode,
184  info.data + i,
185  info.dataSize,
186  &codePoint,
187  &offset) == LE_FAILURE)
188  return LE_FAILURE;
189 
190  i += offset;
191 
192  if(j == idx)
193  return codePoint;
194 
195  j++;
196  }
197 
198  return 0;
199 }

203 int32_t _leTableString_Compare(const leTableString* _this,
204  const leString* tgt)
205 {
206  uint32_t myLen, len, itr, chr;
207 
208  LE_ASSERT_THIS();
209 
210  if(tgt == NULL)
211  return -1;
212 
213  myLen = _this->fn->length(_this);
214  len = tgt->fn->length(tgt);
215 
216  if(myLen < len)
217  {
218  len = myLen;
219  }
220 
221  if(len == 0)
222  return 0;
223 
224  for(itr = 0; itr < len; itr++)
225  {
226  chr = tgt->fn->charAt(tgt, itr);
227 
228  if(_this->fn->charAt(_this, itr) != chr)
229  {
230  return _this->fn->charAt(_this, itr) - chr;
231  }
232  }
233 
234  return 0;
235 }

273 uint32_t _leTableString_ToChar(const leTableString* _this, leChar* buf, uint32_t size)
274 {
275  uint32_t myLen, itr;
276 
277  LE_ASSERT_THIS();
278 
279  if(_this->index == LE_STRING_NULLIDX ||
280  buf == NULL ||
281  size == 0)
282  {
283  return LE_FAILURE;
284  }
285 
286  myLen = _this->fn->length(_this);
287 
288  if(size >= myLen)
289  {
290  size = myLen;
291  }
292 
293  for(itr = 0; itr < size; itr++)
294  {
295  buf[itr] = _this->fn->charAt(_this, itr);
296  }
297 
298  return size;
299 }
