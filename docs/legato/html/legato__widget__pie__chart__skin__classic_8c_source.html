68 static void nextState(lePieChartWidget* chart)
69 {
70  switch(chart->widget.status.drawState)
71  {
72  case NOT_STARTED:
73  {
74  paintState.alpha = 255;
75 
76 #if LE_ALPHA_BLENDING_ENABLED == 1
77  if(chart->fn->getCumulativeAlphaEnabled(chart) == LE_TRUE)
78  {
79  paintState.alpha = chart->fn->getCumulativeAlphaAmount(chart);
80  }
81 #endif
82 
83  if(chart->widget.style.backgroundType != LE_WIDGET_BACKGROUND_NONE)
84  {
85  chart->widget.status.drawState = DRAW_BACKGROUND;
86  chart->widget.drawFunc = (leWidget_DrawFunction_FnPtr)&drawBackground;
87 
88  return;
89  }
90  }
91  // fall through
92  case DRAW_BACKGROUND:
93  {
94  chart->widget.status.drawState = DRAW_PIE_CHART;
95  chart->widget.drawFunc = (leWidget_DrawFunction_FnPtr)&drawPieChart;
96 
97  return;
98  }
99  case DRAW_PIE_CHART:
100  {
101  if(chart->widget.style.borderType != LE_WIDGET_BORDER_NONE)
102  {
103  chart->widget.drawFunc = (leWidget_DrawFunction_FnPtr)&drawBorder;
104  chart->widget.status.drawState = DRAW_BORDER;
105 
106  return;
107  }
108  }
109  // fall through
110  case DRAW_BORDER:
111  {
112  chart->widget.status.drawState = DONE;
113  chart->widget.drawFunc = NULL;
114  }
115  }
116 }

126 static void drawSliceLabel(lePieChartWidget* chart,
127  lePieChartPie* pie)
128 {
129  lePoint centerPoint;
130  int32_t midAngle;
131 
132  leRect textRect, bounds, drawRect;
133 
134  sprintf(strbuff, "%i", (int32_t)pie->value);
135 
136  leFixedString_Constructor(&str, charbuff, MAX_TICK_LABEL_DIGITS);
137  str.fn->setFromCStr(&str, strbuff);
138  str.fn->setFont(&str, chart->labelFont);
139 
140  //Get the string rectangle
141  chart->fn->localRect(chart, &bounds);
142  str.fn->getRect(&str, &textRect);
143 
144  leRectClip(&textRect, &bounds, &drawRect);
145 
146  //Find center angle of pie
147  midAngle = pie->startAngle + (pie->spanAngle / 2);
148 
149  lePolarToXY(pie->radius + pie->offset + chart->labelsOffset, midAngle , &centerPoint);
150  centerPoint.y *= -1;
151 
152  centerPoint.x += chart->widget.rect.x + (chart->widget.rect.width / 2);
153  centerPoint.y += chart->widget.rect.y + (chart->widget.rect.height / 2);
154 
155  drawRect.x = centerPoint.x - (textRect.width / 2);
156  drawRect.y = centerPoint.y - (textRect.height / 2);
157  drawRect.width = textRect.width;
158  drawRect.height = textRect.height;
159 
160  str.fn->_draw(&str,
161  drawRect.x,
162  drawRect.y,
163  LE_HALIGN_CENTER,
164  leScheme_GetRenderColor(chart->widget.scheme, LE_SCHM_TEXT),
165  paintState.alpha);
166 
167  str.fn->destructor(&str);
168 }

170 static void drawSliceArc(lePieChartWidget* chart,
171  lePieChartPie* pie,
172  leColor clr)
173 {
174  uint32_t midAngle;
175  lePoint center, offset;
176  leRect arcRect;
177 
178  center.x = chart->widget.rect.x + paintState.pieCenter.x;
179  center.y = chart->widget.rect.y + paintState.pieCenter.y;
180 
181  arcRect.x = center.x - (pie->radius / 2);
182  arcRect.y = center.y - (pie->radius / 2);
183  arcRect.width = pie->radius;
184  arcRect.height = pie->radius;
185 
186  if (pie->offset != 0)
187  {
188  // find center angle of pie
189  midAngle = pie->startAngle + (pie->spanAngle / 2);
190 
191  // find points of
192  lePolarToXY(pie->offset, midAngle , &offset);
193  offset.y *= -1;
194 
195  arcRect.x += offset.x;
196  arcRect.y += offset.y;
197  }
198 
199  leRenderer_ArcFill(&arcRect,
200  pie->startAngle,
201  pie->spanAngle,
202  pie->radius,
203  LE_FALSE,
204  clr,
205  LE_FALSE,
206  paintState.alpha);
207 }

210 static void drawPieSlice(lePieChartWidget* chart,
211  int32_t idx)
212 {
213  lePieChartPie* pie = leArray_Get(&chart->pieArray, idx);
214  leColor clr;
215 
216  if (pie->scheme != NULL)
217  {
218  clr = leScheme_GetRenderColor(pie->scheme, LE_SCHM_FOREGROUND);
219  }
220  else
221  {
222  clr = leScheme_GetRenderColor( chart->widget.scheme, LE_SCHM_FOREGROUND);
223  }
224 
225  if (pie->radius > 0)
226  {
227  drawSliceArc(chart, pie, clr);
228 
229  if (chart->labelsVisible == LE_TRUE && chart->labelFont != NULL)
230  {
231  drawSliceLabel(chart, pie);
232  }
233  }
234 }

236 static void drawPieChart(lePieChartWidget* chart)
237 {
238  unsigned int i;
239 
240  paintState.pieRect.x = 0;
241  paintState.pieRect.y = 0;
242  paintState.pieRect.width = chart->widget.rect.width;
243  paintState.pieRect.height = chart->widget.rect.height;
244 
245  leUtils_RectToScreenSpace((leWidget*)chart, &paintState.pieRect);
246 
247  paintState.pieCenter.x = paintState.pieRect.width / 2;
248  paintState.pieCenter.y = paintState.pieRect.height / 2;
249 
250  // draw the pies
251  for (i = 0; i < chart->pieArray.size; i++)
252  {
253  drawPieSlice(chart, i);
254  }
255 
256  nextState(chart);
257 }
