278 gfxResult DRV_${ControllerName}_BlitBuffer(int32_t x,
279  int32_t y,
280  gfxPixelBuffer* buf)
281 {
282 
283 <#if BlitBufferFunctionGenerateMode == "Use Bulk Write">
284 <#if DataWriteSize == "8" && PixelDataTxSize8Bit != "2 (Little-Endian)">
285  int row;
286  uint16_t clr;
287 </#if>
288  uint16_t* ptr;
289  uint8_t parm[4];
290 
291  GFX_Disp_Intf intf;
292 
293  if (drv.state != RUN)
294  return GFX_FAILURE;
295 
296  intf = (GFX_Disp_Intf) drv.port_priv;
297 
298 <#if XAddressOffset != 0>
299  //Add X offset
300  x += ${XAddressOffset};
301 </#if>
302 <#if YAddressOffset != 0>
303  //Add Y offset
304  y += ${YAddressOffset};
305 </#if>
306 
307  DRV_${ControllerName}_NCSAssert(intf);
308 
309 <#if FrameAddressSize == "2-bytes">
310  //Write X/Column Address
311  parm[0] = x>>8;
312  parm[1] = x;
313  parm[2] = (x + buf->size.width - 1) >>8;
314  parm[3] = (x + buf->size.width - 1);
315  GFX_Disp_Intf_WriteCommand(intf, 0x${SetXAddressCommand});
316  GFX_Disp_Intf_WriteData(intf, parm, 4);
317 
318  //Write Y/Page Address
319  parm[0] = y>>8;
320  parm[1] = y;
321  parm[2] = (y + buf->size.height - 1)>>8;
322  parm[3] = (y + buf->size.height - 1);
323  GFX_Disp_Intf_WriteCommand(intf, 0x${SetYAddressCommand});
324  GFX_Disp_Intf_WriteData(intf, parm, 4);
325 <#else>
326  //Write X/Column Address
327  parm[0] = x;
328  parm[1] = (x + buf->size.width - 1);
329  GFX_Disp_Intf_WriteCommand(intf, 0x${SetXAddressCommand});
330  GFX_Disp_Intf_WriteData(intf, parm, 2);
331 
332  //Write Y/Page Address
333  parm[0] = y;
334  parm[1] = (y + buf->size.height - 1);
335  GFX_Disp_Intf_WriteCommand(intf, 0x${SetYAddressCommand});
336  GFX_Disp_Intf_WriteData(intf, parm, 2);
337 </#if>
338 
339  //Start Memory Write
340  GFX_Disp_Intf_WriteCommand(intf, 0x${MemoryWriteCommand});
341 
342 <#if DataWriteSize == "16">
343  ptr = gfxPixelBufferOffsetGet_Unsafe(buf, 0, 0);
344  GFX_Disp_Intf_WriteData16(intf, (uint16_t *) ptr, buf->size.width * buf->size.height);
345 <#elseif DataWriteSize == "8">
346 <#if PixelDataTxSize8Bit == "2 (Little-Endian)">
347  ptr = gfxPixelBufferOffsetGet_Unsafe(buf, 0, 0);
348  GFX_Disp_Intf_WriteData(intf,
349  (uint8_t *) ptr,
350  PIXEL_BUFFER_BYTES_PER_PIXEL *
351  buf->size.width *
352 buf->size.height);
353 <#else>
354 
355  for(row = 0; row < buf->size.height; row++)
356  {
357  int col, dataIdx;
358  ptr = gfxPixelBufferOffsetGet_Unsafe(buf, 0, row);
359 <#if PixelDataTxSize8Bit != "2 (Little-Endian)">
360  for(col = 0, dataIdx = 0; col < buf->size.width; col++)
361  {
362  clr = ptr[col];
363 <#if PixelDataTxSize8Bit == "3">
364  pixelBuffer[dataIdx++] = (uint8_t) ((clr & 0xf800) >> 8);
365  pixelBuffer[dataIdx++] = (uint8_t) ((clr & 0x07e0) >> 3 );
366  pixelBuffer[dataIdx++] = (uint8_t) ((clr & 0x001f) << 3);
367 </#if>
368 <#if PixelDataTxSize8Bit == "2 (Normal)">
369  pixelBuffer[dataIdx++] = (uint8_t) (clr >> 8);
370  pixelBuffer[dataIdx++] = (uint8_t) (uint8_t) (clr & 0xff);
371 </#if>
372  }
373 </#if>
374  GFX_Disp_Intf_WriteData(intf,
375  pixelBuffer,
376  PIXEL_BUFFER_BYTES_PER_PIXEL *
377  buf->size.width);
378  }
379 </#if>
380 </#if>
381  DRV_${ControllerName}_NCSDeassert(intf);
382 <#elseif StubGenerateBuildErrorDisable != true>
383 #error "Blit buffer procedure is not complete. Please complete definition of blit function."
384 </#if>
385 
386  return GFX_SUCCESS;
387 }
