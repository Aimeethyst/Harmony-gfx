228 leResult leInitialize(const gfxDisplayDriver* dispDriver,
229  const gfxGraphicsProcessor* gpuDriver)
230 {
231  uint32_t idx;
232  leWidget* root;
233  gfxIOCTLArg_DisplaySize disp;
234 
235  if(_initialized == LE_TRUE)
236  return LE_FAILURE;
237 
238  memset(&_state, 0, sizeof(leState));
239 
240 #if LE_MEMORY_MANAGER_ENABLE == 1
241  if(leMemory_Init() == LE_FAILURE)
242  return LE_FAILURE;
243 #endif
244 
245  if(leEvent_Init() == LE_FAILURE ||
246  leInput_Init() == LE_FAILURE ||
247  leRenderer_Initialize(dispDriver, gpuDriver) == LE_FAILURE)
248  {
249  return LE_FAILURE;
250  }
251 
252 #if LE_DYNAMIC_VTABLES == 1
253  /* initialize virtual function tables */
254  idx = 0;
255 
256  while(vtableFnTable[idx] != NULL)
257  {
258  vtableFnTable[idx]();
259 
260  idx += 1;
261  }
262 #endif
263 
264  leImage_InitDecoders();
265 
266  for(idx = 0; idx < LE_LAYER_COUNT; idx++)
267  {
268  root = &_state.rootWidget[idx];
269 
270  leWidget_Constructor(root);
271 
272  disp.width = 0;
273  disp.height = 0;
274 
275  dispDriver->ioctl(GFX_IOCTL_GET_DISPLAY_SIZE, &disp);
276 
277  root->rect.x = 0;
278  root->rect.y = 0;
279 
280 #if LE_RENDER_ORIENTATION == 0 || LE_RENDER_ORIENTATION == 180
281  root->rect.width = disp.width;
282  root->rect.height = disp.height;
283 #else
284  root->rect.width = disp.height;
285  root->rect.height = disp.width;
286 #endif
287 
288  root->fn->invalidate(root);
289  root->flags |= LE_WIDGET_ISROOT;
290  root->flags |= LE_WIDGET_IGNOREEVENTS;
291  root->flags |= LE_WIDGET_IGNOREPICK;
292 
293  _state.layerStates[idx].colorMode = LE_DEFAULT_COLOR_MODE;
294  }
295 
296  _initialized = LE_TRUE;
297 
298  return LE_SUCCESS;
299 }

354 leResult leUpdate(uint32_t dt)
355 {
356 #if LE_DRIVER_LAYER_MODE == 1
357  uint32_t itr;
358  gfxIOCTLArg_LayerRect layerRect;
359 #endif
360 
361  leEvent_ProcessEvents();
362 
363 #if LE_DRIVER_LAYER_MODE == 1
364  for(itr = 0; itr < LE_LAYER_COUNT; ++itr)
365  {
366  layerRect.base.id = itr;
367  layerRect.x = 0;
368  layerRect.y = 0;
369  layerRect.width = 0;
370  layerRect.height = 0;
371 
372  leGetRenderState()->dispDriver->ioctl(GFX_IOCTL_GET_LAYER_RECT, &layerRect);
373 
374  _state.layerStates[itr].driverPosition.x = layerRect.x;
375  _state.layerStates[itr].driverPosition.y = layerRect.y;
376 
377  _state.rootWidget[itr].fn->setPosition(&_state.rootWidget[itr],
378  0,
379  0);
380 
381 #if LE_RENDER_ORIENTATION == 90 || LE_RENDER_ORIENTATION == 270
382  _state.rootWidget[itr].fn->setSize(&_state.rootWidget[itr],
383  layerRect.height,
384  layerRect.width);
385 #else
386  _state.rootWidget[itr].fn->setSize(&_state.rootWidget[itr],
387  layerRect.width,
388  layerRect.height);
389 #endif
390  }
391 #endif
392 
393 #if LE_STREAMING_ENABLED == 1
394  // there is an active stream in progress, service that to completion
395  // before painting anything else
396  if(_state.activeStream != NULL)
397  {
398  if(_state.activeStream->isDone(_state.activeStream) == LE_TRUE)
399  {
400  _state.activeStream = NULL;
401  }
402  else
403  {
404  _state.activeStream->exec(_state.activeStream);
405 
406  return LE_SUCCESS;
407  }
408  }
409 #endif
410 
411  updateWidgets(dt);
412 
413  leRenderer_Paint();
414 
415  return LE_SUCCESS;
416 }
