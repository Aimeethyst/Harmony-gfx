4  * Copyright (C) 1991-1998, Thomas G. Lane.
5  * Modification developed 2002-2015 by Guido Vollbeding.

7  * For conditions of distribution and use, see the accompanying 'IJG License.pdf' file.
8  *
9  * This file contains a slow-but-accurate integer implementation of the
10  * inverse DCT (Discrete Cosine Transform). In the IJG code, this routine
11  * must also perform dequantization of the input coefficients.
12  *
13  * A 2-D IDCT can be done by 1-D IDCT on each column followed by 1-D IDCT
14  * on each row (or vice versa, but it's more convenient to emit a row at
15  * a time). Direct algorithms are also available, but they are much more
16  * complex and seem not to be any faster when reduced to code.
17  *
18  * This implementation is based on an algorithm described in
19  * C. Loeffler, A. Ligtenberg and G. Moschytz, "Practical Fast 1-D DCT
20  * Algorithms with 11 Multiplications", Proc. Int'l. Conf. on Acoustics,
21  * Speech, and Signal Processing 1989 (ICASSP '89), pp. 988-991.
22  * The primary algorithm described there uses 11 multiplies and 29 adds.
23  * We use their alternate method with 12 multiplies and 32 adds.
24  * The advantage of this method is that no data path contains more than one
25  * multiplication; this allows a very simple and accurate implementation in
26  * scaled fixed-point arithmetic, with a minimal number of shifts.
27  */

34  * This module is specialized to the case DCTSIZE = 8.

37 /*
38  * The scoop on this scaling stuff is as follows:
39  *
40  * Each 1-D IDCT step produces outputs which are a factor of sqrt(N)
41  * larger than the true IDCT outputs. The final outputs are therefore
42  * a factor of N larger than desired; since N=8 this can be cured by
43  * a simple right shift at the end of the algorithm. The advantage of
44  * this arrangement is that we save two multiplications per 1-D IDCT,
45  * because the y0 and y4 inputs need not be divided by sqrt(N).
46  *
47  * We have to do addition and subtraction of the integer inputs, which
48  * is no problem, and multiplication by fractional constants, which is
49  * a problem to do in integer arithmetic. We multiply all the constants
50  * by CONST_SCALE and convert them to integer constants (thus retaining
51  * CONST_BITS bits of precision in the constants). After doing a
52  * multiplication we have to divide the product by CONST_SCALE, with proper
53  * rounding, to produce the correct output. This division can be done
54  * cheaply as a right shift of CONST_BITS bits. We postpone shifting
55  * as long as possible so that partial sums can be added together with
56  * full fractional precision.
57  *
58  * The outputs of the first pass are scaled up by PASS1_BITS bits so that
59  * they are represented to better-than-integral precision. These outputs
60  * require BITS_IN_JSAMPLE + PASS1_BITS + 3 bits; this fits in a 16-bit word
61  * with the recommended scaling. (To scale up 12-bit sample data further, an
62  * intermediate INT32 array would be needed.)
63  *
64  * To avoid overflow of the 32-bit intermediate results in pass 2, we must
65  * have BITS_IN_JSAMPLE + CONST_BITS + PASS1_BITS <= 26. Error analysis
66  * shows that the values given below are the most effective.
67  */
68 #define DCTSIZE 8 /* The basic DCT block is 8x8 samples */
69 #define DCTSIZE2 64 /* DCTSIZE squared; # of elements in a block */
70 #define BITS_IN_JSAMPLE 8
71 #define NO_ZERO_ROW_TEST
72 #define DCTELEM long
73 
74 #define CONST_BITS 13
75 #define PASS1_BITS 2
76 
77 #define INT32 long
78 
79 /* Some C compilers fail to reduce "FIX(constant)" at compile time, thus
80  * causing a lot of useless floating-point operations at run time.
81  * To get around this we use the following pre-calculated constants.
82  * If you change CONST_BITS you may want to add appropriate values.
83  * (With a reasonable C compiler, you can just rely on the FIX() macro...)
84  */
85 
86 #define FIX_0_298631336 ((INT32) 2446) /* FIX(0.298631336) */
87 #define FIX_0_390180644 ((INT32) 3196) /* FIX(0.390180644) */
88 #define FIX_0_541196100 ((INT32) 4433) /* FIX(0.541196100) */
89 #define FIX_0_765366865 ((INT32) 6270) /* FIX(0.765366865) */
90 #define FIX_0_899976223 ((INT32) 7373) /* FIX(0.899976223) */
91 #define FIX_1_175875602 ((INT32) 9633) /* FIX(1.175875602) */
92 #define FIX_1_501321110 ((INT32) 12299) /* FIX(1.501321110) */
93 #define FIX_1_847759065 ((INT32) 15137) /* FIX(1.847759065) */
94 #define FIX_1_961570560 ((INT32) 16069) /* FIX(1.961570560) */
95 #define FIX_2_053119869 ((INT32) 16819) /* FIX(2.053119869) */
96 #define FIX_2_562915447 ((INT32) 20995) /* FIX(2.562915447) */
97 #define FIX_3_072711026 ((INT32) 25172) /* FIX(3.072711026) */

100 /* Multiply an INT32 variable by an INT32 constant to yield an INT32 result.
101  * For 8-bit samples with the recommended scaling, all the variable
102  * and constant values involved are no more than 16 bits wide, so a
103  * 16x16->32 bit multiply can be used instead of a full 32x32 multiply.
104  * For 12-bit samples, a full 32-bit multiplication will be needed.
105  */
106 
107 #define DESCALE(x,n) ((x) + ((long)0x01 << ((n)-1)))>>(n)
108 #define MULTIPLY(var,constant) ((long)(var) * (constant));
109 #define range_limit(x) ((x)<-128)?-128:((x)>127)?127:(x)
110 
111 /* Dequantize a coefficient by multiplying it by the multiplier-table
112  * entry; produce an int result. In this module, both inputs and result
113  * are 16 bits or less, so either int or short multiply will work.
114  */
115 
116 #define DEQUANTIZE(coef,quantval) ((long)(coef) * (quantval))
117

119 /*
120  * Perform dequantization and inverse DCT on one block of coefficients.
121  *
122  * cK represents sqrt(2) * cos(K*pi/16).
123  */
124 
125 void jpeg_idct_islow (short *inbuf, uint16_t *quantptr)
126 {
127  long tmp0, tmp1, tmp2, tmp3;
128  long tmp10, tmp11, tmp12, tmp13;
129  long z1, z2, z3;
130  uint8_t ctr;
131  short *inptr = inbuf, *outptr;
132  DCTELEM * wsptr;
133  DCTELEM workspace[DCTSIZE2]; /* buffers data between passes */
134 
135  wsptr = workspace;
136 
137  /* Pass 1: process columns from input, store into work array.
138  * Note results are scaled up by sqrt(8) compared to a true IDCT;
139  * furthermore, we scale the results by 2**PASS1_BITS.
140  */
141 
142  for (ctr = DCTSIZE; ctr > 0; ctr--) {
143  /* Due to quantization, we will usually find that many of the input
144  * coefficients are zero, especially the AC terms. We can exploit this
145  * by short-circuiting the IDCT calculation for any column in which all
146  * the AC terms are zero. In that case each output is equal to the
147  * DC coefficient (with scale factor as needed).
148  * With typical images and quantization tables, half or more of the
149  * column DCT calculations can be simplified this way.
150  */
151 
152  if (inptr[DCTSIZE*1] == 0 && inptr[DCTSIZE*2] == 0 &&
153  inptr[DCTSIZE*3] == 0 && inptr[DCTSIZE*4] == 0 &&
154  inptr[DCTSIZE*5] == 0 && inptr[DCTSIZE*6] == 0 &&
155  inptr[DCTSIZE*7] == 0) {
156  /* AC terms all zero */
157  long dcval = DEQUANTIZE(inptr[DCTSIZE*0], quantptr[DCTSIZE*0]) << PASS1_BITS;
158 
159  wsptr[DCTSIZE*0] = dcval;
160  wsptr[DCTSIZE*1] = dcval;
161  wsptr[DCTSIZE*2] = dcval;
162  wsptr[DCTSIZE*3] = dcval;
163  wsptr[DCTSIZE*4] = dcval;
164  wsptr[DCTSIZE*5] = dcval;
165  wsptr[DCTSIZE*6] = dcval;
166  wsptr[DCTSIZE*7] = dcval;
167 
168  inptr++; /* advance pointers to next column */
169  quantptr++;
170  wsptr++;
171  continue;
172  }
173 
174  /* Even part: reverse the even part of the forward DCT. */
175  /* The rotator is sqrt(2)*c(-6). */
176 
177  z2 = DEQUANTIZE(inptr[DCTSIZE*2], quantptr[DCTSIZE*2]);
178  z3 = DEQUANTIZE(inptr[DCTSIZE*6], quantptr[DCTSIZE*6]);
179 
180  z1 = MULTIPLY(z2 + z3, FIX_0_541196100);
181  tmp2 = z1 + MULTIPLY(z3, - FIX_1_847759065);
182  tmp3 = z1 + MULTIPLY(z2, FIX_0_765366865);
183 
184  z2 = DEQUANTIZE(inptr[DCTSIZE*0], quantptr[DCTSIZE*0]);
185  z3 = DEQUANTIZE(inptr[DCTSIZE*4], quantptr[DCTSIZE*4]);
186 
187  tmp0 = (z2 + z3) << CONST_BITS;
188  tmp1 = (z2 - z3) << CONST_BITS;
189 
190  tmp10 = tmp0 + tmp3;
191  tmp13 = tmp0 - tmp3;
192  tmp11 = tmp1 + tmp2;
193  tmp12 = tmp1 - tmp2;
194 
195  /* Odd part per figure 8; the matrix is unitary and hence its
196  * transpose is its inverse. i0..i3 are y7,y5,y3,y1 respectively.
197  */
198 
199  tmp0 = DEQUANTIZE(inptr[DCTSIZE*7], quantptr[DCTSIZE*7]);
200  tmp1 = DEQUANTIZE(inptr[DCTSIZE*5], quantptr[DCTSIZE*5]);
201  tmp2 = DEQUANTIZE(inptr[DCTSIZE*3], quantptr[DCTSIZE*3]);
202  tmp3 = DEQUANTIZE(inptr[DCTSIZE*1], quantptr[DCTSIZE*1]);
203 
204  z2 = tmp0 + tmp2;
205  z3 = tmp1 + tmp3;
206 
207  z1 = MULTIPLY(z2 + z3, FIX_1_175875602); /* c3 */
208  z2 = MULTIPLY(z2, - FIX_1_961570560); /* -c3-c5 */
209  z3 = MULTIPLY(z3, - FIX_0_390180644); /* -c3+c5 */
210  z2 += z1;
211  z3 += z1;
212 
213  z1 = MULTIPLY(tmp0 + tmp3, - FIX_0_899976223); /* -c3+c7 */
214  tmp0 = MULTIPLY(tmp0, FIX_0_298631336); /* -c1+c3+c5-c7 */
215  tmp3 = MULTIPLY(tmp3, FIX_1_501321110); /* c1+c3-c5-c7 */
216  tmp0 += z1 + z2;
217  tmp3 += z1 + z3;
218 
219  z1 = MULTIPLY(tmp1 + tmp2, - FIX_2_562915447); /* -c1-c3 */
220  tmp1 = MULTIPLY(tmp1, FIX_2_053119869); /* c1+c3-c5+c7 */
221  tmp2 = MULTIPLY(tmp2, FIX_3_072711026); /* c1+c3+c5-c7 */
222  tmp1 += z1 + z3;
223  tmp2 += z1 + z2;
224 
225  /* Final output stage: inputs are tmp10..tmp13, tmp0..tmp3 */
226 
227  wsptr[DCTSIZE*0] = (long) DESCALE((tmp10 + tmp3), (CONST_BITS-PASS1_BITS));
228  wsptr[DCTSIZE*7] = (long) DESCALE((tmp10 - tmp3), (CONST_BITS-PASS1_BITS));
229  wsptr[DCTSIZE*1] = (long) DESCALE((tmp11 + tmp2), (CONST_BITS-PASS1_BITS));
230  wsptr[DCTSIZE*6] = (long) DESCALE((tmp11 - tmp2), (CONST_BITS-PASS1_BITS));
231  wsptr[DCTSIZE*2] = (long) DESCALE((tmp12 + tmp1), (CONST_BITS-PASS1_BITS));
232  wsptr[DCTSIZE*5] = (long) DESCALE((tmp12 - tmp1), (CONST_BITS-PASS1_BITS));
233  wsptr[DCTSIZE*3] = (long) DESCALE((tmp13 + tmp0), (CONST_BITS-PASS1_BITS));
234  wsptr[DCTSIZE*4] = (long) DESCALE((tmp13 - tmp0), (CONST_BITS-PASS1_BITS));
235 
236  inptr++; /* advance pointers to next column */
237  quantptr++;
238  wsptr++;
239  }
240 
241  /* Pass 2: process rows from work array, store into output array.
242  * Note that we must descale the results by a factor of 8 == 2**3,
243  * and also undo the PASS1_BITS scaling.
244  */
245 
246  wsptr = workspace;
247  outptr = &inbuf[0];
248  for (ctr = 0; ctr < DCTSIZE; ctr++) {
249  /* Rows of zeroes can be exploited in the same way as we did with columns.
250  * However, the column calculation has created many nonzero AC terms, so
251  * the simplification applies less often (typically 5% to 10% of the time).
252  * On machines with very fast multiplication, it's possible that the
253  * test takes more time than it's worth. In that case this section
254  * may be commented out.
255  */
256 
257 #ifndef NO_ZERO_ROW_TEST
258  if (wsptr[1] == 0 && wsptr[2] == 0 && wsptr[3] == 0 && wsptr[4] == 0 &&
259  wsptr[5] == 0 && wsptr[6] == 0 && wsptr[7] == 0) {
260  /* AC terms all zero */
261  JSAMPLE dcval = range_limit[(LONG) DESCALE((INT32) wsptr[0], PASS1_BITS+3)
262  & RANGE_MASK];
263 
264  outptr[0] = dcval;
265  outptr[1] = dcval;
266  outptr[2] = dcval;
267  outptr[3] = dcval;
268  outptr[4] = dcval;
269  outptr[5] = dcval;
270  outptr[6] = dcval;
271  outptr[7] = dcval;
272 
273  wsptr += DCTSIZE; /* advance pointer to next row */
274  continue;
275  }
276 #endif
277 
278  /* Even part: reverse the even part of the forward DCT. */
279  /* The rotator is sqrt(2)*c(-6). */
280 
281  z2 = (INT32) wsptr[2];
282  z3 = (INT32) wsptr[6];
283 
284  z1 = MULTIPLY(z2 + z3, FIX_0_541196100);
285  tmp2 = z1 + MULTIPLY(z3, - FIX_1_847759065);
286  tmp3 = z1 + MULTIPLY(z2, FIX_0_765366865);
287 
288  tmp0 = ((INT32) wsptr[0] + (INT32) wsptr[4]) << CONST_BITS;
289  tmp1 = ((INT32) wsptr[0] - (INT32) wsptr[4]) << CONST_BITS;
290 
291  tmp10 = tmp0 + tmp3;
292  tmp13 = tmp0 - tmp3;
293  tmp11 = tmp1 + tmp2;
294  tmp12 = tmp1 - tmp2;
295 
296  /* Odd part per figure 8; the matrix is unitary and hence its
297  * transpose is its inverse. i0..i3 are y7,y5,y3,y1 respectively.
298  */
299 
300  tmp0 = (INT32) wsptr[7];
301  tmp1 = (INT32) wsptr[5];
302  tmp2 = (INT32) wsptr[3];
303  tmp3 = (INT32) wsptr[1];
304 
305  z2 = tmp0 + tmp2;
306  z3 = tmp1 + tmp3;
307 
308  z1 = MULTIPLY(z2 + z3, FIX_1_175875602); /* c3 */
309  z2 = MULTIPLY(z2, - FIX_1_961570560); /* -c3-c5 */
310  z3 = MULTIPLY(z3, - FIX_0_390180644); /* -c3+c5 */
311  z2 += z1;
312  z3 += z1;
313 
314  z1 = MULTIPLY(tmp0 + tmp3, - FIX_0_899976223); /* -c3+c7 */
315  tmp0 = MULTIPLY(tmp0, FIX_0_298631336); /* -c1+c3+c5-c7 */
316  tmp3 = MULTIPLY(tmp3, FIX_1_501321110); /* c1+c3-c5-c7 */
317  tmp0 += z1 + z2;
318  tmp3 += z1 + z3;
319 
320  z1 = MULTIPLY(tmp1 + tmp2, - FIX_2_562915447); /* -c1-c3 */
321  tmp1 = MULTIPLY(tmp1, FIX_2_053119869); /* c1+c3-c5+c7 */
322  tmp2 = MULTIPLY(tmp2, FIX_3_072711026); /* c1+c3+c5-c7 */
323  tmp1 += z1 + z3;
324  tmp2 += z1 + z2;
325 
326  /* Final output stage: inputs are tmp10..tmp13, tmp0..tmp3 */
327 
328  outptr[0] = (short)range_limit((long) DESCALE(tmp10 + tmp3, CONST_BITS+PASS1_BITS+3));
329  outptr[7] = (short)range_limit((long) DESCALE(tmp10 - tmp3, CONST_BITS+PASS1_BITS+3));
330  outptr[1] = (short)range_limit((long) DESCALE(tmp11 + tmp2, CONST_BITS+PASS1_BITS+3));
331  outptr[6] = (short)range_limit((long) DESCALE(tmp11 - tmp2, CONST_BITS+PASS1_BITS+3));
332  outptr[2] = (short)range_limit((long) DESCALE(tmp12 + tmp1, CONST_BITS+PASS1_BITS+3));
333  outptr[5] = (short)range_limit((long) DESCALE(tmp12 - tmp1, CONST_BITS+PASS1_BITS+3));
334  outptr[3] = (short)range_limit((long) DESCALE(tmp13 + tmp0, CONST_BITS+PASS1_BITS+3));
335  outptr[4] = (short)range_limit((long) DESCALE(tmp13 - tmp0, CONST_BITS+PASS1_BITS+3));
336 
337  outptr += DCTSIZE; /* advance pointer to next row */
338  wsptr += DCTSIZE; /* advance pointer to next row */
339  }
340 }
