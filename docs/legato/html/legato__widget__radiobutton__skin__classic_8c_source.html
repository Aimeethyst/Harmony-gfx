64 void _leRadioButtonWidget_GetImageRect(const leRadioButtonWidget* btn,
65  leRect* imgRect,
66  leRect* imgSrcRect)
67 {
68  leRect textRect;
69  leRect bounds;
70 
71  btn->fn->localRect(btn, &bounds);
72 
73  imgRect->x = 0;
74  imgRect->y = 0;
75 
76  if(btn->string != NULL)
77  {
78  btn->string->fn->getRect(btn->string, &textRect);
79  }
80  else
81  {
82  textRect = leRect_Zero;
83  }
84 
85  // calculate image dimensions
86  if(btn->selected == LE_TRUE)
87  {
88  if(btn->selectedImage != NULL)
89  {
90  imgRect->width = btn->selectedImage->buffer.size.width;
91  imgRect->height = btn->selectedImage->buffer.size.height;
92  }
93  else
94  {
95  imgRect->width = btn->circleButtonSize;
96  imgRect->height = btn->circleButtonSize;
97  }
98  }
99  else
100  {
101  if(btn->unselectedImage != NULL)
102  {
103  imgRect->width = btn->unselectedImage->buffer.size.width;
104  imgRect->height = btn->unselectedImage->buffer.size.height;
105  }
106  else
107  {
108  imgRect->width = btn->circleButtonSize;
109  imgRect->height = btn->circleButtonSize;
110  }
111  }
112 
113  *imgSrcRect = *imgRect;
114 
115  leUtils_ArrangeRectangle(imgRect,
116  textRect,
117  bounds,
118  btn->widget.style.halign,
119  btn->widget.style.valign,
120  btn->imagePosition,
121  btn->widget.margin.left,
122  btn->widget.margin.top,
123  btn->widget.margin.right,
124  btn->widget.margin.bottom,
125  btn->imageMargin);
126 
127  leRectClipAdj(imgRect, &bounds, imgSrcRect, imgRect);
128 
129  // move the rect to layer space
130  leUtils_RectToScreenSpace((leWidget*)btn, imgRect);
131 }

205 static void nextState(leRadioButtonWidget* btn)
206 {
207  switch(btn->widget.status.drawState)
208  {
209  case NOT_STARTED:
210  {
211  paintState.alpha = 255;
212 
213 #if LE_ALPHA_BLENDING_ENABLED == 1
214  if(btn->fn->getCumulativeAlphaEnabled(btn) == LE_TRUE)
215  {
216  paintState.alpha = btn->fn->getCumulativeAlphaAmount(btn);
217  }
218 #endif
219 
220  if(btn->widget.style.backgroundType != LE_WIDGET_BACKGROUND_NONE)
221  {
222  btn->widget.status.drawState = DRAW_BACKGROUND;
223  btn->widget.drawFunc = (leWidget_DrawFunction_FnPtr)&drawBackground;
224 
225  return;
226  }
227  }
228  // fall through
229  case DRAW_BACKGROUND:
230  {
231  btn->widget.status.drawState = DRAW_IMAGE;
232  btn->widget.drawFunc = (leWidget_DrawFunction_FnPtr)&drawImage;
233 
234  return;
235  }
236  case DRAW_IMAGE:
237  {
238  if(btn->string != NULL)
239  {
240  btn->widget.status.drawState = DRAW_STRING;
241  btn->widget.drawFunc = (leWidget_DrawFunction_FnPtr)&drawString;
242 
243  return;
244  }
245  }
246  // fall through
247  case DRAW_STRING:
248  {
249  if(btn->widget.style.borderType != LE_WIDGET_BORDER_NONE)
250  {
251  btn->widget.drawFunc = (leWidget_DrawFunction_FnPtr)&drawBorder;
252  btn->widget.status.drawState = DRAW_BORDER;
253 
254  return;
255  }
256  }
257  // fall through
258  case DRAW_BORDER:
259  {
260  btn->widget.status.drawState = DONE;
261  btn->widget.drawFunc = NULL;
262  }
263  }
264 }

433 static void drawImage(leRadioButtonWidget* btn)
434 {
435  leRect imgRect, imgSrcRect;
436  const leImage* img = NULL;
437 
438  _leRadioButtonWidget_GetImageRect(btn, &imgRect, &imgSrcRect);
439 
440  if(btn->selected == LE_FALSE)
441  {
442  img = btn->unselectedImage;
443  }
444  else
445  {
446  img = btn->selectedImage;
447  }
448 
449  if(img == NULL)
450  {
451  if (btn->circleButtonSize < MIN_CIRCLE_SIZE)
452  {
453  drawCircle(btn, &imgRect, btn->selected);
454  }
455  else
456  {
457  drawCircleArcs(btn, &imgRect, btn->selected);
458  }
459  }
460  else
461  {
462  leImage_Draw(img,
463  &imgSrcRect,
464  imgRect.x,
465  imgRect.y,
466  paintState.alpha);
467 
468 #if LE_STREAMING_ENABLED == 1
469  if(leGetActiveStream() != NULL)
470  {
471  leGetActiveStream()->onDone = onImageStreamFinished;
472  leGetActiveStream()->userData = btn;
473 
474  btn->widget.status.drawState = WAIT_IMAGE;
475 
476  return;
477  }
478 #endif
479  }
480 
481  nextState(btn);
482 }

539 void _leRadioButtonWidget_Paint(leRadioButtonWidget* btn)
540 {
541  if(btn->widget.status.drawState == NOT_STARTED)
542  {
543  nextState(btn);
544  }
545 
546 #if LE_STREAMING_ENABLED == 1
547  if(btn->widget.status.drawState == WAIT_STRING ||
548  btn->widget.status.drawState == WAIT_IMAGE)
549  {
550  return;
551  }
552 #endif
553 
554  while(btn->widget.status.drawState != DONE)
555  {
556  btn->widget.drawFunc((leWidget*)btn);
557 
558 #if LE_PREEMPTION_LEVEL == 2
559  break;
560 #endif
561 
562 #if LE_STREAMING_ENABLED == 1
563  if(btn->widget.status.drawState == WAIT_STRING ||
564  btn->widget.status.drawState == WAIT_IMAGE)
565  break;
566 #endif
567  }
568 }
