109 gfxResult DRV_GFX2D_Fill(gfxPixelBuffer* dest,
110  const gfxRect* clipRect,
111  const gfxColor color)
112 {
113  GFX2D_BUFFER dest_buffer;
114  GFX2D_RECTANGLE dest_rect;
115 
116  if(dest->orientation != GFX_ORIENT_0)
117  return GFX_FAILURE;
118 
119  dest_buffer.width = dest->size.width;
120  dest_buffer.height = dest->size.height;
121  dest_buffer.format = gfx2dFormats[dest->mode];
122  dest_buffer.dir = GFX2D_XY00;
123  dest_buffer.addr = (uint32_t)dest->pixels;
124 
125  dest_rect.x = (uint32_t)clipRect->x;
126  dest_rect.y = (uint32_t)clipRect->y;
127  dest_rect.width = (uint32_t)clipRect->width;
128  dest_rect.height = (uint32_t)clipRect->height;
129 
130  PLIB_GFX2D_Fill(&dest_buffer, &dest_rect, color);
131 
132  /* Wait for instruction to complete */
133  while ( PLIB_GFX2D_GetGlobalStatusBusy() == true ) ;
134  //while (gpu_end == 0) {
135  //};
136 
137  return GFX_SUCCESS;
138 }

140 gfxResult DRV_GFX2D_Blit(const gfxPixelBuffer* source,
141  const gfxRect* srcRect,
142  const gfxPixelBuffer* dest,
143  const gfxRect* destRect)
144 {
145  GFX2D_BUFFER dest_buffer;
146  GFX2D_RECTANGLE dest_rect;
147  GFX2D_BUFFER src_buffer;
148  GFX2D_RECTANGLE src_rect;
149 
150  if(source->orientation != GFX_ORIENT_0 ||
151  dest->orientation != GFX_ORIENT_0)
152  return GFX_FAILURE;
153 
154  dest_buffer.width = dest->size.width;
155  dest_buffer.height = dest->size.height;
156  dest_buffer.format = gfx2dFormats[dest->mode];
157  dest_buffer.dir = GFX2D_XY00;
158  dest_buffer.addr = (uint32_t)dest->pixels;
159 
160  dest_rect.x = (uint32_t)destRect->x;
161  dest_rect.y = (uint32_t)destRect->y;
162  dest_rect.width = (uint32_t)destRect->width;
163  dest_rect.height = (uint32_t)destRect->height;
164 
165  src_buffer.width = source->size.width;
166  src_buffer.height = source->size.height;
167  src_buffer.format = gfx2dFormats[source->mode];
168  src_buffer.dir = GFX2D_XY00;
169  src_buffer.addr = (uint32_t)source->pixels;
170 
171  src_rect.x = (uint32_t)srcRect->x;
172  src_rect.y = (uint32_t)srcRect->y;
173  src_rect.width = (uint32_t)srcRect->width;
174  src_rect.height = (uint32_t)srcRect->height;
175 
176  if ( blendState == GFX_BLEND_NONE )
177  {
178  PLIB_GFX2D_Copy(&dest_buffer, &dest_rect, &src_buffer, &src_rect);
179  }
180  else
181  {
182  GFX2D_BLEND blend;
183 
184  blend.spe=GFX2D_SPE_MULTIPLY;
185  blend.func= GFX2D_FUNCTION_SPE;
186  blend.dfact= GFX2D_FACTOR_DST_COLOR;
187  blend.sfact= GFX2D_FACTOR_SRC_COLOR;
188 
189  DRV_GFX2D_Blend(&dest_buffer, (GFX2D_RECTANGLE*)&dest_rect, &dest_buffer,
190  (GFX2D_RECTANGLE*)&dest_rect, &src_buffer, (GFX2D_RECTANGLE*)&src_rect,
191  blend);
192  }
193 
194  /* Wait for instruction to complete */
195  while ( PLIB_GFX2D_GetGlobalStatusBusy() == true ) ;
196  //while (gpu_end == 0) {
197  //};
198 
199  return GFX_SUCCESS;
200 }
