74 static void nextState(leCircularGaugeWidget* gauge)
75 {
76  switch(gauge->widget.status.drawState)
77  {
78  case NOT_STARTED:
79  {
80  paintState.alpha = 255;
81 
82 #if LE_ALPHA_BLENDING_ENABLED == 1
83  if(gauge->fn->getCumulativeAlphaEnabled(gauge) == LE_TRUE)
84  {
85  paintState.alpha = gauge->fn->getCumulativeAlphaAmount(gauge);
86  }
87 #endif
88 
89  if(gauge->widget.style.backgroundType != LE_WIDGET_BACKGROUND_NONE)
90  {
91  gauge->widget.status.drawState = DRAW_BACKGROUND;
92  gauge->widget.drawFunc = (leWidget_DrawFunction_FnPtr)&drawBackground;
93 
94  return;
95  }
96  }
97  // fall through
98  case DRAW_BACKGROUND:
99  {
100  gauge->widget.status.drawState = DRAW_CIRCULAR_GAUGE;
101  gauge->widget.drawFunc = (leWidget_DrawFunction_FnPtr)&drawCircularGauge;
102 
103  return;
104  }
105  case DRAW_CIRCULAR_GAUGE:
106  {
107  if(gauge->widget.style.borderType != LE_WIDGET_BORDER_NONE)
108  {
109  gauge->widget.drawFunc = (leWidget_DrawFunction_FnPtr)&drawBorder;
110  gauge->widget.status.drawState = DRAW_BORDER;
111 
112  return;
113  }
114  }
115  // fall through
116  case DRAW_BORDER:
117  {
118  gauge->widget.status.drawState = DONE;
119  gauge->widget.drawFunc = NULL;
120  }
121  }
122 }

133  const leCircularGaugeArc* arc)
134 {

136 
137  int32_t startAngle = leDegreesFromPercent(arc->startAngle, gauge->centerAngle, gauge->startAngle);
138  int32_t endAngle = leDegreesFromPercent(arc->endAngle, gauge->centerAngle, gauge->startAngle);
139 
140  leColor clr = leScheme_GetRenderColor(arc->scheme,
141  LE_SCHM_FOREGROUND);
142 
143  rect.x = paintState.gaugeOrigin.x - (gauge->radius + arc->offset);
144  rect.y = paintState.gaugeOrigin.y - (gauge->radius + arc->offset);
145  rect.width = (gauge->radius + arc->offset) * 2;
146  rect.height = (gauge->radius + arc->offset) * 2;

149  startAngle,
150  endAngle - startAngle,
151  arc->thickness,
152  LE_FALSE,
153  clr,
154  LE_FALSE,
155  paintState.alpha);

176 static void _paintTickRange(const leCircularGaugeWidget* gauge,
177  const leCircularGaugeTickRange* rng)
178 {
179  leColor clr = leScheme_GetRenderColor(rng->scheme, LE_SCHM_FOREGROUND);
180 
181  int32_t startAngle = leDegreesFromPercent(rng->startAngle, gauge->centerAngle, gauge->startAngle);
182  int32_t endAngle = leDegreesFromPercent(rng->endAngle, gauge->centerAngle, gauge->startAngle);
183  int32_t t;
184 
185  if(endAngle < startAngle)
186  {
187  t = endAngle;
188  endAngle = startAngle;
189  startAngle = t;
190  }
191 
192  if(rng->thickness == 0)
193  return;
194 
195  if(rng->divisions == 0)
196  {
197  //draw tick at start
198  _paintTicksAtAngle(rng->offset,
199  rng->thickness,
200  startAngle,
201  clr);
202 
203  //draw tick at end
204  _paintTicksAtAngle(rng->offset,
205  rng->thickness,
206  endAngle,
207  clr);
208  }
209  else
210  {
211  int32_t increment = (rng->endAngle - rng->startAngle) / rng->divisions;
212  uint32_t idx;
213 
214  //draw start and middle ticks
215  for(idx = 0; idx < rng->divisions; ++idx)
216  {
217  _paintTicksAtAngle(rng->offset,
218  rng->thickness,
219  leDegreesFromPercent(rng->startAngle + (idx * increment), gauge->centerAngle, gauge->startAngle),
220  clr);
221  }
222 
223  //draw tick at end
224  _paintTicksAtAngle(rng->offset,
225  rng->thickness,
226  endAngle,
227  clr);
228  }
229 }

252

254  stringRect.y = paintState.gaugeOrigin.y + (-lblPoint.y) - stringRect.height / 2;

257  stringRect.x,

264 static void _paintLabelRange(const leCircularGaugeWidget* gauge,
265  const leCircularGaugeLabelRange* rng)
266 {
267  leColor clr;
268 
269  if(rng->font == NULL)
270  return;
271 
272  clr = leScheme_GetRenderColor(rng->scheme, LE_SCHM_TEXT);
273 
274  if(rng->divisions == 0)
275  {
276  // draw label at start
277  _paintLabelValueAtAngle(leDegreesFromPercent(rng->startAngle, gauge->centerAngle, gauge->startAngle),
278  rng->offset,
279  0,
280  clr,
281  rng->font);
282 
283  // draw label at end
284  _paintLabelValueAtAngle(leDegreesFromPercent(rng->endAngle, gauge->centerAngle, gauge->startAngle),
285  rng->offset,
286  100,
287  clr,
288  rng->font);
289  }
290  else
291  {
292  if(gauge->centerAngle < 0)
293  {
294  int32_t startAngle = leDegreesFromPercent(rng->startAngle, gauge->centerAngle, gauge->startAngle);
295  int32_t endAngle = leDegreesFromPercent(rng->endAngle, gauge->centerAngle, gauge->startAngle);
296  int32_t increment = -((startAngle - endAngle) / rng->divisions);
297  uint32_t idx;
298 
299  // draw start and middle ticks
300  for(idx = 0; idx < rng->divisions; ++idx)
301  {
302  int32_t angle = startAngle + (idx * increment);
303 
304  uint32_t deltaVal = lePercentWholeRounded(idx, rng->divisions);
305 
306  _paintLabelValueAtAngle(angle,
307  rng->offset,
308  deltaVal,
309  clr,
310  rng->font);
311  }
312 
313  // draw label at end
314  _paintLabelValueAtAngle(leDegreesFromPercent(rng->endAngle, gauge->centerAngle, gauge->startAngle),
315  rng->offset,
316  100,
317  clr,
318  rng->font);
319  }
320  else
321  {
322  int32_t startAngle = leDegreesFromPercent(rng->startAngle, gauge->centerAngle, gauge->startAngle);
323  int32_t endAngle = leDegreesFromPercent(rng->endAngle, gauge->centerAngle, gauge->startAngle);
324  int32_t increment = (endAngle - startAngle) / rng->divisions;
325  uint32_t idx;
326 
327  // draw start and middle ticks
328  for(idx = 0; idx < rng->divisions; ++idx)
329  {
330  int32_t angle = startAngle + (idx * increment);
331 
332  uint32_t deltaVal = lePercentWholeRounded(idx, rng->divisions);
333 
334  _paintLabelValueAtAngle(angle,
335  rng->offset,
336  deltaVal,
337  clr,
338  rng->font);
339  }
340 
341  // draw label at end
342  _paintLabelValueAtAngle(leDegreesFromPercent(rng->endAngle, gauge->centerAngle, gauge->startAngle),
343  rng->offset,
344  100,
345  clr,
346  rng->font);
347  }
348  }

351 static void paintCenterCircle(const leCircularGaugeWidget* gauge)
352 {
353  leRect circleRect;
354 
355  if(gauge->centerCircleVisible == LE_FALSE)
356  return;
357 
358  uint32_t thickness = gauge->centerCircleThickness;
359 
360  if(thickness > gauge->centerCircleRadius)
361  {
362  thickness = gauge->centerCircleRadius;
363  }
364 
365  uint32_t circleRadius = lePercentOf(gauge->radius, gauge->centerCircleRadius);
366 
367  circleRect.x = paintState.gaugeOrigin.x - circleRadius;
368  circleRect.y = paintState.gaugeOrigin.y - circleRadius;
369  circleRect.width = circleRadius * 2;
370  circleRect.height = circleRadius * 2;
371 
372  leRenderer_CircleDraw(&circleRect,
373  thickness,
374  leScheme_GetRenderColor(gauge->widget.scheme, LE_SCHM_FOREGROUND),
375  paintState.alpha);
376 }

378 static void paintHand(const leCircularGaugeWidget* gauge)
379 {
380  lePoint startPt, endPt;
381  int32_t handLength;
382  int32_t angle;
383 
384  if(gauge->handVisible == LE_FALSE)
385  return;
386 
387  handLength = lePercentOf(gauge->radius, gauge->handRadius);
388  angle = leDegreesFromPercent(gauge->value, gauge->centerAngle, gauge->startAngle);
389 
390  if(gauge->centerCircleVisible == LE_FALSE)
391  {
392  startPt.x = 0;
393  startPt.y = 0;
394  }
395  else
396  {
397  lePolarToXY(gauge->centerCircleRadius - 2, angle, &startPt);
398  lePolarToXY(handLength, angle, &endPt);
399  }
400 
401  leRenderer_DrawLine(paintState.gaugeOrigin.x + startPt.x,
402  paintState.gaugeOrigin.y - startPt.y,
403  paintState.gaugeOrigin.x + endPt.x,
404  paintState.gaugeOrigin.y - endPt.y,
405  leScheme_GetRenderColor(gauge->fn->getScheme(gauge), LE_SCHM_FOREGROUND),
406  paintState.alpha);

410 static void drawCircularGauge(leCircularGaugeWidget* gauge)
411 {
412  uint32_t itr;
413  leCircularGaugeArc* arc;
414  leCircularGaugeTickRange* tickRange;
415  leCircularGaugeLabelRange* labelRange;
416 
417  paintState.radius = gauge->radius;
418 
419  paintState.gaugeOrigin.x = gauge->fn->getWidth(gauge)/2;
420  paintState.gaugeOrigin.y = gauge->fn->getHeight(gauge)/2;
421 
422  leUtils_PointToScreenSpace((leWidget*)gauge, &paintState.gaugeOrigin);
423 
424  paintState.gaugeRect.x = paintState.gaugeOrigin.x - gauge->radius;
425  paintState.gaugeRect.y = paintState.gaugeOrigin.y - gauge->radius;
426  paintState.gaugeRect.width = gauge->radius * 2;
427  paintState.gaugeRect.height = gauge->radius * 2;
428 
429  // draw the arcs
430  for(itr = 0; itr < gauge->arcsArray.size; ++itr)
431  {
432  arc = (leCircularGaugeArc*)leArray_Get(&gauge->arcsArray, itr);
433 
434  if(arc->enabled == LE_FALSE)
435  continue;
436 
437  paintArc(gauge, arc);
438  }
439 
440  // draw the tick ranges
441  for(itr = 0; itr < gauge->ticksArray.size; ++itr)
442  {
443  tickRange = (leCircularGaugeTickRange*)leArray_Get(&gauge->ticksArray, itr);
444 
445  if(tickRange->enabled == LE_FALSE)
446  continue;
447 
448  _paintTickRange(gauge, tickRange);
449  }
450 
451  // draw the labels
452  for(itr = 0; itr < gauge->labelsArray.size; ++itr)
453  {
454  labelRange = (leCircularGaugeLabelRange*)leArray_Get(&gauge->labelsArray, itr);
455 
456  if(labelRange->enabled == LE_FALSE)
457  continue;
458 
459  _paintLabelRange(gauge, labelRange);
460  }
461 
462  // draw the hand
463  paintHand(gauge);
464 
465  // draw center circle
466  paintCenterCircle(gauge);
467 
468  //leRenderer_RectLine(&paintState.gaugeRect,
469  // leColorValue(LE_COLOR_RED, leRenderer_CurrentColorMode()),
470  // 255);
471 
472  nextState(gauge);
473 }
