64  leRect* imgRect,
65  leRect* imgSrcRect)
66 {
67  leRect textRect = leRect_Zero;
68  leRect bounds;
69 
70  btn->fn->localRect(btn, &bounds);
71 
72  imgRect->x = 0;
73  imgRect->y = 0;
74 
75  if(btn->string != NULL)
76  {
77  btn->string->fn->getRect(btn->string, &textRect);
78  }
79 
80  //leString_GetMultiLineRect(&btn->string, &textRect, -1);
81 
82  if(btn->state != LE_BUTTON_STATE_UP)
83  {
84  if(btn->pressedImage != NULL)
85  {
86  imgRect->width = btn->pressedImage->buffer.size.width;
87  imgRect->height = btn->pressedImage->buffer.size.height;
88  }
89  }
90  else
91  {
92  if(btn->releasedImage != NULL)
93  {
94  imgRect->width = btn->releasedImage->buffer.size.width;
95  imgRect->height = btn->releasedImage->buffer.size.height;
96  }
97  }
98 
99  *imgSrcRect = *imgRect;
100 
101  leUtils_ArrangeRectangle(imgRect,
102  textRect,
103  bounds,
104  btn->widget.style.halign,
105  btn->widget.style.valign,
106  btn->imagePosition,
107  btn->widget.margin.left,
108  btn->widget.margin.top,
109  btn->widget.margin.right,
110  btn->widget.margin.bottom,
111  btn->imageMargin);
112 
113  if(btn->state != LE_BUTTON_STATE_UP)
114  {
115  imgRect->x += btn->pressedOffset;
116  imgRect->y += btn->pressedOffset;
117  }
118 
119  leRectClipAdj(imgRect, &bounds, imgSrcRect, imgRect);
120 
121  // move the rect to screen space
122  leUtils_RectToScreenSpace((leWidget*)btn, imgRect);

185 void _leButtonWidget_InvalidateBorderAreas(const leButtonWidget* btn)
186 {
187  leRect rect, dmgRect;
188  int32_t left, top, right, bottom;
189 
190  if(btn->widget.style.borderType == LE_WIDGET_BORDER_NONE)
191  return;
192 
193  btn->fn->rectToScreen(btn, &rect);
194 
195  if(btn->widget.style.borderType == LE_WIDGET_BORDER_LINE)
196  {
197  if(rect.width == 0 || rect.height == 0)
198  return;
199 
200  if(rect.width <= 1 || rect.height <= 1)
201  {
202  btn->fn->_damageArea(btn, &rect);
203 
204  return;
205  }
206 
207  left = 1;
208  top = 1;
209  right = 1;
210  bottom = 1;
211  }
212  else
213  {
214  if(rect.width == 0 || rect.height == 0)
215  return;
216 
217  if(rect.width <= 3 || rect.height <= 3)
218  {
219  btn->fn->_damageArea(btn, &rect);
220 
221  return;
222  }
223 
224  right = 2;
225  bottom = 2;
226 
227  if(btn->state == LE_BUTTON_STATE_UP)
228  {
229  left = 1;
230  top = 1;
231  }
232  else
233  {
234  left = 2;
235  top = 2;
236  }
237  }
238 
239  // left line
240  dmgRect.x = rect.x;
241  dmgRect.y = rect.y;
242  dmgRect.width = left;
243  dmgRect.height = rect.height;
244 
245  btn->fn->_damageArea(btn, &dmgRect);
246 
247  // top line
248  dmgRect.width = rect.width;
249  dmgRect.height = top;
250 
251  btn->fn->_damageArea(btn, &dmgRect);
252 
253  // right line
254  dmgRect.x = rect.x + rect.width - right;
255  dmgRect.y = rect.y;
256  dmgRect.width = right;
257  dmgRect.height = rect.height;
258 
259  btn->fn->_damageArea(btn, &dmgRect);
260 
261  // bottom line
262  dmgRect.x = rect.x;
263  dmgRect.y = rect.y + rect.height - bottom;
264  dmgRect.width = rect.width;
265  dmgRect.height = bottom;
266 
267  btn->fn->_damageArea(btn, &dmgRect);
268 }

274 static void nextState(leButtonWidget* btn)
275 {
276  switch(btn->widget.status.drawState)
277  {
278  case NOT_STARTED:
279  {
280  paintState.alpha = 255;
281 
282 #if LE_ALPHA_BLENDING_ENABLED == 1
283  if(btn->fn->getCumulativeAlphaEnabled(btn) == LE_TRUE)
284  {
285  paintState.alpha = btn->fn->getCumulativeAlphaAmount(btn);
286  }
287 #endif
288 
289  if(btn->widget.style.backgroundType != LE_WIDGET_BACKGROUND_NONE)
290  {
291  btn->widget.status.drawState = DRAW_BACKGROUND;
292  btn->widget.drawFunc = (leWidget_DrawFunction_FnPtr)&drawBackground;
293 
294  return;
295  }
296  }
297  // fall through
298  case DRAW_BACKGROUND:
299  {
300  if((btn->state != LE_BUTTON_STATE_UP && btn->pressedImage != NULL) ||
301  (btn->state == LE_BUTTON_STATE_UP && btn->releasedImage != NULL))
302  {
303  btn->widget.status.drawState = DRAW_IMAGE;
304  btn->widget.drawFunc = (leWidget_DrawFunction_FnPtr)&drawImage;
305 
306  return;
307  }
308  }
309  // fall through
310  case DRAW_IMAGE:
311  {
312  if(btn->string != NULL && btn->string->fn->isEmpty(btn->string) == LE_FALSE)
313  {
314  btn->widget.status.drawState = DRAW_STRING;
315  btn->widget.drawFunc = (leWidget_DrawFunction_FnPtr)&drawString;
316 
317  return;
318  }
319  }
320  // fall through
321  case DRAW_STRING:
322  {
323  if(btn->widget.style.borderType != LE_WIDGET_BORDER_NONE)
324  {
325  btn->widget.drawFunc = (leWidget_DrawFunction_FnPtr)&drawBorder;
326  btn->widget.status.drawState = DRAW_BORDER;
327 
328  return;
329  }
330  }
331  // fall through
332  case DRAW_BORDER:
333  {
334  btn->widget.status.drawState = DONE;
335  btn->widget.drawFunc = NULL;
336  }
337  }
338 }

478 void _leButtonWidget_Paint(leButtonWidget* btn)
479 {
480  if(btn->widget.status.drawState == NOT_STARTED)
481  {
482  nextState(btn);
483  }
484 
485 #if LE_STREAMING_ENABLED == 1
486  if(btn->widget.status.drawState == WAIT_IMAGE ||
487  btn->widget.status.drawState == WAIT_STRING)
488  {
489  return;
490  }
491 #endif
492 
493  while(btn->widget.status.drawState != DONE)
494  {
495  btn->widget.drawFunc((leWidget*)btn);
496 
497 #if LE_PREEMPTION_LEVEL == 2
498  break;
499 #endif
500 
501 #if LE_STREAMING_ENABLED == 1
502  if(btn->widget.status.drawState == WAIT_IMAGE ||
503  btn->widget.status.drawState == WAIT_STRING)
504  break;
505 #endif
506  }
507 }
